## js comb sort

### Solution

```javascript
function combSort(array) {
	let inc = array.length
	let swap = true

	while (!inc == 1 || swap) {
		// recalibrate inc
		inc = Math.floor(inc / 1.3)
		if (inc < 1) {
			inc = 1
		}

		swap = false
		for (let i = 0; i < array.length - inc; i++) {
			if (array[i] > array[i + inc]) {
				// swap position
				let tmp = array[i]
				array[i] = array[i + inc]
				array[i + inc] = tmp
				swap = true
			}
		}
	}
	return array
}
```
